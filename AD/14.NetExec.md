# 🤖 Automated Multi-Protocol Scan Script

## 📜 run_nxc_scan.sh

    #!/bin/bash

    IP_LIST="ips.txt"
    LOG_DIR="tmp/nxc_scan_$(date +%Y%m%d_%H%M%S)"
    mkdir -p "$LOG_DIR"

    PROTOCOLS=("smb" "wmi" "rdp" "winrm" "ldap" "nfs" "vnc" "mssql" "ssh" "ftp")

    if ! command -v nxc &> /dev/null; then
        echo "Error: nxc command not found."
        exit 1
    fi

    while IFS= read -r TARGET || [ -n "$TARGET" ]; do
        [ -z "$TARGET" ] && continue

        echo "Scanning $TARGET..."

        for PROTOCOL in "${PROTOCOLS[@]}"; do
            LOG_FILE="${LOG_DIR}/nxc_${PROTOCOL}_${TARGET}.log"
            nxc "$PROTOCOL" "$TARGET" &> "$LOG_FILE"
        done
    done < "$IP_LIST"

    echo "All scans completed. Logs stored in $LOG_DIR"

---

## ▶️ Usage

    chmod +x run_nxc_scan.sh
    ./run_nxc_scan.sh
