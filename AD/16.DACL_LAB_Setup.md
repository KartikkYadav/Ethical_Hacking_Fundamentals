  ## First add Users in the active directory :

- Run the below command in DC1 computer

       # -----------------------------------------------
      # FULL AD LAB SETUP
      # - Creates OU
      # - Creates users (including attacker jdoe)
      # - Configures WinRM access
      # - Adds realism groups
      # -----------------------------------------------

      Import-Module ActiveDirectory
 
      try {
       # ===============================================
      # 1. Get Domain Info
      # ===============================================
      $domain = Get-ADDomain -ErrorAction Stop
      $domainDN = $domain.DistinguishedName
      $dnsRoot = $domain.DNSRoot
 
      Write-Host "[+] Domain detected: $dnsRoot"

      # ===============================================
      # 2. Create OU
      # ===============================================
      $ouName = "LabUsers"
      $ouDN = "OU=$ouName,$domainDN"

      if (-not (Get-ADOrganizationalUnit -Filter "Name -eq '$ouName'" -ErrorAction SilentlyContinue)) {
        New-ADOrganizationalUnit -Name $ouName -Path $domainDN
        Write-Host "[+] Created OU: $ouName"
      }
      else {
        Write-Host "[=] OU already exists: $ouName"
      }

      # ===============================================
      # 3. Define Users
      # ===============================================
      $users = @(
        "alice","bob","charlie","david","emma",
        "frank","grace","harry","irene","jack",
        "kate","leo","mia","nina","oscar",
        "peter","quinn","rose","steve","jdoe"
      )

      $defaultPassword = ConvertTo-SecureString "  User1234" -AsPlainText -Force

      # ===============================================
      # 4. Create Users
      # ===============================================
      foreach ($username in $users) {

        if (-not (Get-ADUser -Filter "SamAccountName -eq '$username'" -ErrorAction SilentlyContinue)) {

            New-ADUser `
                -Name $username `
                -SamAccountName $username `
                -UserPrincipalName "$username@$dnsRoot" `
                -Path $ouDN `
                -AccountPassword $defaultPassword `
                -Enabled $true `
                -PasswordNeverExpires $true `
                -ChangePasswordAtLogon $false

            Write-Host "[+] Created user: $username"
        }
        else {
            Write-Host "[=] User already exists: $username"
        }
      }

      # ===============================================
      # 5. Add jdoe to Remote Management Users (WinRM)
      # ===============================================
      try {
        Add-ADGroupMember -Identity "Remote Management Users" -Members "jdoe" -ErrorAction Stop
        Write-Host "[+] jdoe added to Remote Management Users (domain group)"
      }
      catch {
        Write-Host "[!] Domain group not found. Trying local group..."

        try {
            net localgroup "Remote Management Users" jdoe /add
            Write-Host "[+] jdoe added to local Remote Management Users"
        }
        catch {
            Write-Host "[=] jdoe already in Remote Management Users or failed silently"
        }
      }

      # ===============================================
      # 6. Create Realistic Groups
      # ===============================================
      $groups = @("Helpdesk", "IT-Support", "HR", "Finance")
  
      foreach ($group in $groups) {
        if (-not (Get-ADGroup -Filter "Name -eq '$group'" -ErrorAction SilentlyContinue)) {
            New-ADGroup -Name $group -GroupScope Global -Path $ouDN
            Write-Host "[+] Created group: $group"
        }
        else {
            Write-Host "[=] Group already exists: $group"
        }
      }

      # ===============================================
      # 7. Populate Groups (Realism)
      # ===============================================
      try {
        Add-ADGroupMember "Helpdesk" -Members "alice","bob"
        Add-ADGroupMember "IT-Support" -Members "charlie","david"
        Add-ADGroupMember "HR" -Members "emma","grace"
        Add-ADGroupMember "Finance" -Members "frank","harry"

        Write-Host "[+] Users assigned to groups"
      }
      catch {
          Write-Host "[=] Some users already assigned to groups"
      }

      # ===============================================
      # 8. Enable WinRM (for remote exploitation)
      # ===============================================
      try {
        Enable-PSRemoting -Force
        Write-Host "[+] WinRM enabled"
      }
      catch {
        Write-Host "[=] WinRM already enabled"
      }

      # ===============================================
      #   9. Final Output
      # ===============================================
      Write-Host ""
      Write-Host "=====================================" -ForegroundColor Green
      Write-Host " AD LAB SETUP COMPLETED SUCCESSFULLY " -ForegroundColor Green
      Write-Host "=====================================" -ForegroundColor Green
      Write-Host "Attacker: jdoe / User1234"
      Write-Host "Domain  : $dnsRoot"
      Write-Host ""
      }
        catch {
      Write-Host "[-] Fatal Error: $_" -ForegroundColor Red
      }

       # ===============================================
    # Lab 1: Grant WriteMembers + WinRM access to user "alice"
    # ===============================================

    Import-Module ActiveDirectory

    try {
    $username = "alice"

    # ===============================================
    # 1. Get User & Domain Admins Group
    # ===============================================
    $user = Get-ADUser $username -ErrorAction Stop
    $group = Get-ADGroup "Domain Admins" -ErrorAction Stop

    $userSID = New-Object System.Security.Principal.SecurityIdentifier($user.SID)
    $groupDN = $group.DistinguishedName

    $groupDE = [ADSI]"LDAP://$groupDN"
    $acl = $groupDE.ObjectSecurity

    # ===============================================
    # 2. Grant WriteProperty (WriteMembers) on 'member'
    # ===============================================
    $memberGUID = [Guid]"bf9679c0-0de6-11d0-a285-00aa003049e2"

    $ace = New-Object System.DirectoryServices.ActiveDirectoryAccessRule(
        $userSID,
        [System.DirectoryServices.ActiveDirectoryRights]::WriteProperty,
        [System.Security.AccessControl.AccessControlType]::Allow,
        $memberGUID
    )

    $acl.AddAccessRule($ace)
    $groupDE.ObjectSecurity = $acl
    $groupDE.CommitChanges()

    Write-Host "[+] $username can now modify Domain Admins membership (WriteMembers)"

    # ===============================================
    # 3. Add user to Remote Management Users
    # ===============================================
    try {
        Add-ADGroupMember -Identity "Remote Management Users" -Members $username -ErrorAction Stop
        Write-Host "[+] $username added to Remote Management Users (domain group)"
    }
    catch {
        Write-Host "[!] Domain group not found. Trying local group..."

        try {
            net localgroup "Remote Management Users" $username /add
            Write-Host "[+] $username added to local Remote Management Users"
        }
        catch {
            Write-Host "[-] Failed to add $username to Remote Management Users"
        }
    }

    Write-Host "`n✅ Lab 1 completed successfully!" -ForegroundColor Green
    }
