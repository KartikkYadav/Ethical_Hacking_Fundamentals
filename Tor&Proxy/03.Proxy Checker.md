# Proxy Checker (Async Python)

## Ultra-Fast Proxy Checker (Async Python)

An asynchronous proxy checker built with **asyncio + aiohttp** that can test **thousands of proxies per minute**.

### Features

- HTTP proxies
- HTTPS proxies
- SOCKS4 proxies
- SOCKS5 proxies
- IP verification
- High concurrency scanning

# Ultra Fast Proxy Checker Script

## Create File

    vim proxy-checker.py

---

## Code

```python
import asyncio
import aiohttp
from aiohttp_socks import ProxyConnector
from aiohttp import ClientTimeout

INPUT_FILE = "proxies.txt"
OUTPUT_FILE = "working.txt"
CONCURRENCY = 500
TEST_URL = "https://api.ipify.org?format=json"

semaphore = asyncio.Semaphore(CONCURRENCY)

async def check_proxy(proxy):
    proxy_types = [
        ("http", f"http://{proxy}"),
        ("https", f"https://{proxy}"),
        ("socks4", f"socks4://{proxy}"),
        ("socks5", f"socks5://{proxy}")
    ]

    for proto, proxy_url in proxy_types:
        try:
            async with semaphore:
                connector = ProxyConnector.from_url(proxy_url)
                timeout = ClientTimeout(total=6)

                async with aiohttp.ClientSession(
                    connector=connector,
                    timeout=timeout
                ) as session:

                    async with session.get(TEST_URL) as r:
                        if r.status == 200:
                            data = await r.json()
                            ip = data.get("ip")

                            ip_addr, port = proxy.split(":")

                            print(f"[WORKING] {proto}://{ip_addr}:{port} -> {ip}")

                            with open(OUTPUT_FILE, "a") as f:
                                f.write(f"{proto}://{ip_addr}:{port}\n")

                            return
        except:
            pass


async def main():
    with open(INPUT_FILE) as f:
        proxies = [line.strip() for line in f if line.strip()]

    print(f"Loaded {len(proxies)} proxies\n")

    tasks = [check_proxy(p) for p in proxies]
    await asyncio.gather(*tasks)


if __name__ == "__main__":
    asyncio.run(main())
```
